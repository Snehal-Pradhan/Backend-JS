```js
app.get('/users', async (req, res) => {
  const query = {};

  if (req.query.role) query.role = req.query.role;
  if (req.query.age_gte) query.age = { $gte: parseInt(req.query.age_gte) };

  const users = await User.find(query);
  res.json(users);
});
```
